# 1. Instalar dependências
pip install facebook-business google-cloud-bigquery pandas python-dotenv

# 2. Configurar credenciais
cp .env.example .env
# edite o .env com suas credenciais reais

# 3. Autenticar no Google Cloud (para BigQuery)
gcloud auth application-default login

# 4. Rodar — exportando para CSV primeiro (para testar)
python meta_ads_extractor.py --dest csv

# 5. Quando tudo estiver ok, carregar no BigQuery
python meta_ads_extractor.py --dest bigquery

# 6. Especificar período personalizado
python meta_ads_extractor.py --start 2026-01-01 --end 2026-03-01 --dest bigquery
